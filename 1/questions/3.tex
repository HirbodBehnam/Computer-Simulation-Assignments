\begin{enumerate}
    \item شبیه‌سازی درخواست‌هایی که برای یک node فرستاده می‌شود یا شبیه سازی توپولوژی شبکه برای اینکه چک کنیم به هم دیگر دسترسی دارند \lr{node}ها یا خیر
    \item لود سرور‌ و زمان پاسخ آن مورد بررسی قرار می‌گیرد
    \item شبیه‌سازی زمان بندی تسک‌ها در دستگاه
\end{enumerate}
